package com.example.myapplication;

import android.os.Bundle;
import android.widget.CompoundButton;
import android.widget.ImageView;
import android.widget.ToggleButton;

import androidx.appcompat.app.AppCompatActivity;

public class MainActivity extends AppCompatActivity {

    ImageView imageBulb;

    ToggleButton toggleButton;

    @Override
    protected void onCreate(Bundle savedInstanceState) {

        super.onCreate(savedInstanceState);

        setContentView(R.layout.activity_main);

        imageBulb =
                findViewById(R.id.imageBulb);

        toggleButton =
                findViewById(R.id.toggleButton);

        toggleButton.setOnCheckedChangeListener(
                new CompoundButton.OnCheckedChangeListener() {

            @Override
            public void onCheckedChanged(
                    CompoundButton buttonView,
                    boolean isChecked
            ) {

                if(isChecked) {

                    imageBulb.setImageResource(
                            R.drawable.bulbon
                    );
                }

                else {

                    imageBulb.setImageResource(
                            R.drawable.bulboff
                    );
                }
            }
        });
    }
}

MainActivity.java
package com.example.myapplication;

import android.graphics.Color;
import android.graphics.Typeface;
import android.os.Bundle;
import android.view.View;
import android.widget.Button;
import android.widget.CheckBox;
import android.widget.EditText;
import android.widget.RadioButton;
import android.widget.TextView;

import androidx.appcompat.app.AppCompatActivity;

public class MainActivity extends AppCompatActivity {

    EditText etName;

    RadioButton rbRed, rbBlue;

    CheckBox cbBold, cbItalic;

    Button btnShow, btnClear;

    TextView txtResult;

    @Override
    protected void onCreate(Bundle savedInstanceState) {

        super.onCreate(savedInstanceState);

        setContentView(R.layout.activity_main);

        etName = findViewById(R.id.etName);

        rbRed = findViewById(R.id.rbRed);
        rbBlue = findViewById(R.id.rbBlue);

        cbBold = findViewById(R.id.cbBold);
        cbItalic = findViewById(R.id.cbItalic);

        btnShow = findViewById(R.id.btnShow);
        btnClear = findViewById(R.id.btnClear);

        txtResult = findViewById(R.id.txtResult);

        btnShow.setOnClickListener(
                new View.OnClickListener() {

            @Override
            public void onClick(View v) {

                txtResult.setText(
                        etName.getText().toString()
                );

                if(rbRed.isChecked()) {

                    txtResult.setTextColor(
                            Color.RED
                    );
                }

                if(rbBlue.isChecked()) {

                    txtResult.setTextColor(
                            Color.BLUE
                    );
                }

                if(cbBold.isChecked()) {

                    txtResult.setTypeface(
                            null,
                            Typeface.BOLD
                    );
                }

                if(cbItalic.isChecked()) {

                    txtResult.setTypeface(
                            null,
                            Typeface.ITALIC
                    );
                }
            }
        });

        btnClear.setOnClickListener(
                new View.OnClickListener() {

            @Override
            public void onClick(View v) {

                etName.setText("");

                txtResult.setText("");
            }
        });
    }
}
